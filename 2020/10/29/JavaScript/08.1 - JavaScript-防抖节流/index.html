    let last, timer
    return function (args) {
      let _this = this
      let _args = args
      let now = +new Date()
      if (last && now < last + delay) {
        clearTimeout(timer)
        timer = setTimeout(function () {
          last = now
          fun.apply(_this, _args)
        }, delay)
      } else {
        last = now
        fun.apply(_this, _args)
      }
    }
  }
